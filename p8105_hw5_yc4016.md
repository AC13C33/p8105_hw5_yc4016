norm_mean_p = function(n=30, mu, sigma = 5) {
  norm_data = tibble(x = rnorm(n, mean = mu, sd = sigma))
  norm_data %>% 
    t.test (mu = 0, alpha = 0.05, conf.level = 0.95) %>% 
    broom::tidy ()
}
